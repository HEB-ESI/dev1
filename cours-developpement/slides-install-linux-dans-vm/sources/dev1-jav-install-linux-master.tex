\begin{frame}{Définition}
	\begin{quote}
		\og{}
		Un système d'exploitation (Operating System ou OS)
		est un ensemble de programmes spécialisés qui permet 
		l'utilisation des ressources matérielles d'un ou plusieurs ordinateurs. 
		\fg{}
		{\footnotesize{\color{gray}source: universalis}}
	\end{quote}	
\end{frame}

\begin{frame}
	Il assure :
	\begin{itemize}
	\item le démarrage (Boot) de l'ordinateur
	\item l'exécution des logiciels applicatifs
	\end{itemize} 
	Il remplit deux fonctions majeures :
	\begin{itemize}
	\item la gestion des ressources matérielles 
	\item la fourniture de services aux applications
	\\(appels système offrant une interface de plus haut niveau que celle 	
	de la machine physique) 
	\end{itemize}
	\begin{center}
		\includegraphics[width=2cm]{../img/run.jpg}
		\qquad
		\includegraphics[width=2cm]{../img/ram.jpg}
		\qquad
		\includegraphics[width=2cm]{../img/harddisk.jpg}
	\end{center}
	
	\note{
		\begin{itemize}
		\item Permet de lancer un programme
		\item Gère l'accès CPU (ordonnancement) 
		\item Permet la communication entre programmes
		\item Gère la mémoire (lié au point précédent)
		\item Crée la notion de système de fichiers
		\item Gère les périphériques (souris, écran, clé USB)
		\end{itemize}
	}
\end{frame}

\begin{frame}{Les modes d'installation}
	Un OS peut s'installer
	\begin{itemize}
	\item en \emph{single-boot}
	\item en \emph{multi-boot}
	\item dans une machine virtuelle
	\end{itemize}
	Certains peuvent aussi se lancer en \emph{live}
\end{frame}
